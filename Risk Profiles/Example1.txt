p1 <- seq(0,0.99,length=100)
p2 <- p1
i1 <- 3
i2 <- 9
f <- function(p1,p2){(i2*p1^2*p2^2-i2*p1^2*p2+(-2)*i2*p1*p2^2+3*i2*p1*p2-i2*p1+i2*p2^2-2*i2*p2+i2+2*i1*p1*p2-i1*p1-i1*p1^2*p2+i1-i1*p2)/(p1^2*p2^2-p1^2*p2+(-2)*p1*p2^2+3*p1*p2-p1+p2^2-2*p2+1)}
risk <- outer(p1,p2,f)
op <- par(bg="white")
persp(p1, p2, risk, theta =60, phi = 30, expand = 0.5, col = "lightblue")
